    def _creduce(self):

        B1, B2, P, W = self.B1, self.B2, self.P, self.W

        return P_sigma(1, 0) << B2 << (P + cid(1)) << P_sigma(1, 0) << B1 << (W + cid(1))

    @property

        return self.creduce().space
